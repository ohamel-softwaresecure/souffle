54  * Read and return the next tuple.
55  *
56  * Returns nullptr if no tuple was readable.
57  * @return
58  */
59  std::unique_ptr<RamDomain[]> readNextTuple() override {
60  if (file.eof()) {
61  return nullptr;
62  }
63  std::string line;
64  std::unique_ptr<RamDomain[]> tuple = std::make_unique<RamDomain[]>(typeAttributes.size());
65 
66  if (!getline(file, line)) {
67  return nullptr;
68  }
69  // Handle Windows line endings on non-Windows systems
70  if (!line.empty() && line.back() == '\r') {
71  line = line.substr(0, line.length() - 1);
72  }
73  ++lineNumber;
74 
75  size_t start = 0;
76  size_t end = 0;
77  size_t columnsFilled = 0;
78  for (uint32_t column = 0; columnsFilled < arity; column++) {
79  std::string element = nextElement(line, start, end);
80  if (inputMap.count(column) == 0) {
81  continue;
82  }
83  ++columnsFilled;
84 
85  try {
86  switch (typeAttributes.at(inputMap[column])[0]) {
87  case 's':
88  tuple[inputMap[column]] = symbolTable.unsafeLookup(element);
89  break;
90  case 'r':
91  tuple[inputMap[column]] = readRecord(element, typeAttributes[inputMap[column]]);
92  break;
93  case 'i':
94  tuple[inputMap[column]] = RamDomainFromString(element);
95  break;
96  case 'u':
97  tuple[inputMap[column]] = ramBitCast(RamUnsignedFromString(element));
98  break;
99  case 'f':
100  tuple[inputMap[column]] = ramBitCast(RamFloatFromString(element));
101  break;
102  default:
103  assert(false && "Invalid type attribute");
104  }
105  } catch (...) {
106  std::stringstream errorMessage;
107  errorMessage << "Error converting <" + element + "> in column " << column + 1 << " in line "
108  << lineNumber << "; ";
109  throw std::invalid_argument(errorMessage.str());
110  }
111  }
112 
113  return tuple;
114  }

116  std::string nextElement(const std::string& line, size_t& start, size_t& end) {
117  std::string element;
118 
119  // Handle record/tuple delimiter coincidence.
120  if (delimiter.find(',') != std::string::npos) {
121  int record_parens = 0;
122  size_t next_delimiter = line.find(delimiter, start);
123 
124  // Find first delimiter after the record.
125  while (end < std::min(next_delimiter, line.length()) || record_parens != 0) {
126  // Track the number of parenthesis.
127  if (line[end] == '[') {
128  ++record_parens;
129  } else if (line[end] == ']') {
130  --record_parens;
131  }
132 
133  // Check for unbalanced parenthesis.
134  if (record_parens < 0) {
135  break;
136  };
137 
138  ++end;
139 
140  // Find a next delimiter if the old one is invalid.
141  // But only if inside the unbalance parenthesis.
142  if (end == next_delimiter && record_parens != 0) {
143  next_delimiter = line.find(delimiter, end);
144  }
145  }
146 
147  // Handle the end-of-the-line case where parenthesis are unbalanced.
148  if (record_parens != 0) {
149  std::stringstream errorMessage;
150  errorMessage << "Unbalanced record parenthesis " << lineNumber << "; ";
151  throw std::invalid_argument(errorMessage.str());
152  }
153  } else {
154  end = std::min(line.find(delimiter, start), line.length());
155  }
156 
157  // Check for missing value.
158  if (start > end) {
159  std::stringstream errorMessage;
160  errorMessage << "Values missing in line " << lineNumber << "; ";
161  throw std::invalid_argument(errorMessage.str());
162  }
163 
164  element = line.substr(start, end - start);
165  start = end + delimiter.size();
166 
167  return element;
168  }

177  std::map<int, int> getInputColumnMap(const IODirectives& ioDirectives, const unsigned arity_) const {
178  std::string columnString = "";
179  if (ioDirectives.has("columns")) {
180  columnString = ioDirectives.get("columns");
181  }
182  std::map<int, int> inputColumnMap;
183 
184  if (!columnString.empty()) {
185  std::istringstream iss(columnString);
186  std::string mapping;
187  int index = 0;
188  while (std::getline(iss, mapping, ':')) {
189  inputColumnMap[stoi(mapping)] = index++;
190  }
191  if (inputColumnMap.size() < arity_) {
192  throw std::invalid_argument("Invalid column set was given: <" + columnString + ">");
193  }
194  } else {
195  while (inputColumnMap.size() < arity_) {
196  int size = static_cast<int>(inputColumnMap.size());
197  inputColumnMap[size] = size;
198  }
199  }
200  return inputColumnMap;
201  }
