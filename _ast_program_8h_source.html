54  void print(std::ostream& os) const override {
55  /* Print types */
56  os << "// ----- Types -----\n";
57  for (const auto& cur : types) {
58  os << *cur.second << "\n";
59  }
60 
61  /* Print components */
62  if (!components.empty()) {
63  os << "\n// ----- Components -----\n";
64  for (const auto& cur : components) {
65  os << *cur << "\n";
66  }
67  }
68 
69  /* Print instantiations */
70  if (!instantiations.empty()) {
71  os << "\n";
72  for (const auto& cur : instantiations) {
73  os << *cur << "\n";
74  }
75  }
76 
77  /* Print functors */
78  os << "\n// ----- Functors -----\n";
79  for (const auto& cur : functors) {
80  const std::unique_ptr<AstFunctorDeclaration>& f = cur.second;
81  os << "\n\n// -- " << f->getName() << " --\n";
82  f->print(os);
83  os << "\n";
84  }
85 
86  /* Print relations */
87  os << "\n// ----- Relations -----\n";
88  for (const auto& cur : relations) {
89  const std::unique_ptr<AstRelation>& rel = cur.second;
90  os << "\n\n// -- " << rel->getName() << " --\n";
91  os << *rel << "\n\n";
92  for (const auto clause : rel->getClauses()) {
93  os << *clause << "\n\n";
94  }
95  }
96 
97  if (!clauses.empty()) {
98  os << "\n// ----- Orphan Clauses -----\n";
99  os << join(clauses, "\n\n", print_deref<std::unique_ptr<AstClause>>()) << "\n";
100  }
101 
102  if (!loads.empty()) {
103  os << "\n// ----- Orphan Load directives -----\n";
104  os << join(loads, "\n\n", print_deref<std::unique_ptr<AstLoad>>()) << "\n";
105  }
106  if (!stores.empty()) {
107  os << "\n// ----- Orphan Store directives -----\n";
108  os << join(stores, "\n\n", print_deref<std::unique_ptr<AstStore>>()) << "\n";
109  }
110 
111  if (!pragmaDirectives.empty()) {
112  os << "\n// ----- Pragma -----\n";
113  for (const auto& cur : pragmaDirectives) {
114  os << *cur << "\n";
115  }
116  }
117  }

240  AstProgram* clone() const override {
241  auto res = new AstProgram();
242 
243  for (const auto& cur : pragmaDirectives) {
244  res->pragmaDirectives.emplace_back(cur->clone());
245  }
246  for (const auto& cur : components) {
247  res->components.emplace_back(cur->clone());
248  }
249  for (const auto& cur : instantiations) {
250  res->instantiations.emplace_back(cur->clone());
251  }
252  for (const auto& cur : types) {
253  res->types.insert(std::make_pair(cur.first, std::unique_ptr<AstType>(cur.second->clone())));
254  }
255  for (const auto& cur : functors) {
256  res->functors.insert(
257  std::make_pair(cur.first, std::unique_ptr<AstFunctorDeclaration>(cur.second->clone())));
258  }
259  for (const auto& cur : relations) {
260  res->relations.insert(
261  std::make_pair(cur.first, std::unique_ptr<AstRelation>(cur.second->clone())));
262  }
263  for (const auto& cur : clauses) {
264  res->clauses.emplace_back(cur->clone());
265  }
266  for (const auto& cur : loads) {
267  res->loads.emplace_back(cur->clone());
268  }
269  for (const auto& cur : printSizes) {
270  res->printSizes.emplace_back(cur->clone());

274  }

276  // TODO (b-scholz): that is odd - revisit!

456  /** finishing parsing */
457  void finishParsing() {
458  // unbound clauses with no relation defined
459  std::vector<std::unique_ptr<AstClause>> unbound;
460 
461  // add clauses
462  for (auto& cur : clauses) {
463  auto pos = relations.find(cur->getHead()->getName());
464  if (pos != relations.end()) {
465  pos->second->addClause(std::move(cur));
466  } else {
467  unbound.push_back(std::move(cur));
468  }
469  }
470  // remember the remaining orphan clauses
471  clauses.clear();
472  clauses.swap(unbound);
473 
474  // unbound directives with no relation defined
475  std::vector<std::unique_ptr<AstLoad>> unboundLoads;
476  std::vector<std::unique_ptr<AstStore>> unboundStores;
477  }
