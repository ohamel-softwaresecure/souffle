282  }

284  AstIntrinsicFunctor(FunctorOp function, std::vector<std::unique_ptr<AstArgument>> operands)
285  : AstFunctor(std::move(operands)), function(function) {
286  assert(isValidFunctorOpArity(function, args.size()) && "invalid number of arguments for functor");
287  }

289  void print(std::ostream& os) const override {
290  if (isInfixFunctorOp(function)) {
291  os << "(";
292  os << join(args, getSymbolForFunctorOp(function), print_deref<std::unique_ptr<AstArgument>>());
293  os << ")";
294  } else {
295  os << getSymbolForFunctorOp(function);
296  os << "(";

493 /**
494  * An argument aggregating a value from a sub-query.
495  * TODO (b-scholz): fix body literal interface;
496  * remove getters/setters for individual literals
497  */
498 class AstAggregator : public AstArgument {
499 public:
500  /**
501  * The kind of utilised aggregation operator.
502  * Note: lower-case is utilized due to a collision with
503  * constants in the parser.
504  */
505  enum Op { min, max, count, sum };
506 
507  /** Creates a new aggregation node */
508  AstAggregator(Op fun) : fun(fun), expression(nullptr) {}
509 
510  void print(std::ostream& os) const override {
511  switch (fun) {
512  case sum:
513  os << "sum";
514  break;
515  case min:
516  os << "min";
517  break;
518  case max:
519  os << "max";
520  break;
521  case count:
522  os << "count";
523  break;
524  default:
525  break;
526  }
527  if (expression) {
528  os << " " << *expression;
529  }
530  os << " : ";
531  if (body.size() > 1) {
532  os << "{ ";
533  }
534  os << join(body, ", ", print_deref<std::unique_ptr<AstLiteral>>());
535  if (body.size() > 1) {
536  os << " }";
537  }
538  }
539 
540  /** Get aggregate operator */
541  Op getOperator() const {
542  return fun;
543  }
544 
545  /** Set target expression */
546  void setTargetExpression(std::unique_ptr<AstArgument> arg) {
547  expression = std::move(arg);
548  }
549 
550  /** Get target expression */
551  const AstArgument* getTargetExpression() const {
552  return expression.get();
553  }
554 
555  /** Get body literals */
556  std::vector<AstLiteral*> getBodyLiterals() const {
557  return toPtrVector(body);
558  }
559 
560  /** Clear body literals */
561  void clearBodyLiterals() {
562  body.clear();
563  }
564 
565  /** Add body literal */
566  void addBodyLiteral(std::unique_ptr<AstLiteral> lit) {
567  body.push_back(std::move(lit));
568  }
569 
570  std::vector<const AstNode*> getChildNodes() const override {
571  auto res = AstArgument::getChildNodes();
572  if (expression) {
573  res.push_back(expression.get());
574  }
575  for (auto& cur : body) {
576  res.push_back(cur.get());
577  }
578  return res;
579  }
580 
581  AstAggregator* clone() const override {
582  auto res = new AstAggregator(fun);
583  res->expression = (expression) ? std::unique_ptr<AstArgument>(expression->clone()) : nullptr;
584  for (const auto& cur : body) {
585  res->body.emplace_back(cur->clone());
586  }
587  res->setSrcLoc(getSrcLoc());
588  return res;
589  }
590 
591  void apply(const AstNodeMapper& map) override {
592  if (expression) {
593  expression = map(std::move(expression));
594  }
595  for (auto& cur : body) {
596  cur = map(std::move(cur));
597  }
598  }
599 
600 protected:
