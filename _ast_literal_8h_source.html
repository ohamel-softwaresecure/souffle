43 /**
44  * Subclass of Literal that represents the use of a relation
45  * either in the head or in the body of a Clause, e.g., parent(x,y).
46  * The arguments of the atom can be variables or constants.
47  */
48 class AstAtom : public AstAtomLiteral {
49 public:
50  AstAtom(AstRelationIdentifier name = AstRelationIdentifier()) : name(std::move(name)) {}
51 
52  AstAtom(AstRelationIdentifier name, std::vector<std::unique_ptr<AstArgument>> args, SrcLocation srcLoc)
53  : name(std::move(name)), arguments(std::move(args)) {
54  setSrcLoc(srcLoc);
55  }
56 
57  /** Return the name of this atom */
58  // TODO (b-scholz): rename to getIdent
59  const AstRelationIdentifier& getName() const {
60  return name;
61  }
62 
63  /** Return the arity of the atom */
64  size_t getArity() const {
65  return arguments.size();
66  }
67 
68  /** Set atom name */
69  void setName(const AstRelationIdentifier& n) {
70  name = n;
71  }
72 
73  /** Returns this class as the referenced atom */
74  const AstAtom* getAtom() const override {
75  return this;
76  }
77 
78  /** Add argument to the atom */
79  void addArgument(std::unique_ptr<AstArgument> arg) {
80  arguments.push_back(std::move(arg));
81  }
82 
83  /** Provides access to the list of arguments of this atom */
84  std::vector<AstArgument*> getArguments() const {
85  return toPtrVector(arguments);
86  }
87 
88  void print(std::ostream& os) const override {
89  os << getName() << "(";
90 
91  for (size_t i = 0; i < arguments.size(); ++i) {
92  if (i != 0) {
93  os << ",";
94  }
95  if (arguments[i] != nullptr) {
96  arguments[i]->print(os);
97  } else {
98  os << "_";
99  }
100  }
101  os << ")";
102  }
103 
104  AstAtom* clone() const override {
105  auto res = new AstAtom(name);
106  res->setSrcLoc(getSrcLoc());
107  for (const auto& cur : arguments) {
108  res->arguments.emplace_back(cur->clone());
109  }
110  return res;
111  }
112 
113  void apply(const AstNodeMapper& map) override {
114  for (auto& arg : arguments) {
115  arg = map(std::move(arg));
116  }
117  }
118 
119  std::vector<const AstNode*> getChildNodes() const override {
120  std::vector<const AstNode*> res;
121  for (auto& cur : arguments) {
122  res.push_back(cur.get());
123  }
124  return res;
125  }
126 
127 protected:
128  bool equal(const AstNode& node) const override {
129  assert(nullptr != dynamic_cast<const AstAtom*>(&node));
130  const auto& other = static_cast<const AstAtom&>(node);
131  return name == other.name && equal_targets(arguments, other.arguments);
132  }
133 
134  /** Name of the atom */
135  AstRelationIdentifier name;
136 
137  /** Arguments of the atom */
138  std::vector<std::unique_ptr<AstArgument>> arguments;
139 };
