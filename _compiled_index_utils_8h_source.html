81 // -------- generic tuple comparator ----------
82 
83 template <unsigned... Columns>
84 struct comparator;
85 
86 template <unsigned First, unsigned... Rest>
87 struct comparator<First, Rest...> {
88  template <typename T>
89  int operator()(const T& a, const T& b) const {
90  return (a[First] < b[First]) ? -1 : ((a[First] > b[First]) ? 1 : comparator<Rest...>()(a, b));
91  }
92  template <typename T>
93  bool less(const T& a, const T& b) const {
94  return a[First] < b[First] || (a[First] == b[First] && comparator<Rest...>().less(a, b));
95  }
96  template <typename T>
97  bool equal(const T& a, const T& b) const {
98  return a[First] == b[First] && comparator<Rest...>().equal(a, b);
99  }
100 };
