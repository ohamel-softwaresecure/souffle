32 /**
33  * A component type is
34  *
35  * name < Type1, Type2, ... >
36  *
37  * where name is the component name and < Type, Type, ... > is a
38  * list of component type parameters (either actual or formal).
39  */
40 class AstComponentType : public AstNode {
41 public:
42  AstComponentType(
43  std::string name = "", std::vector<AstTypeIdentifier> params = std::vector<AstTypeIdentifier>())
44  : name(std::move(name)), typeParams(std::move(params)) {}
45 
46  void print(std::ostream& os) const override {
47  os << name;
48  if (!typeParams.empty()) {
49  os << "<" << join(typeParams, ",") << ">";
50  }
51  }
52 
53  /** get component name */
54  const std::string& getName() const {
55  return name;
56  }
57 
58  /** set component name */
59  void setName(const std::string& n) {
60  name = n;
61  }
62 
63  /** get component type parameters */
64  const std::vector<AstTypeIdentifier>& getTypeParameters() const {
65  return typeParams;
66  }
67 
68  /** set component type parameters */
69  void setTypeParameters(const std::vector<AstTypeIdentifier>& params) {
70  typeParams = params;
71  }
72 
73  AstComponentType* clone() const override {
74  auto* res = new AstComponentType(name, typeParams);
75  res->setSrcLoc(getSrcLoc());
76  return res;
77  }
78 
79 protected:
80  bool equal(const AstNode& node) const override {
81  assert(nullptr != dynamic_cast<const AstComponentType*>(&node));
82  const auto& other = static_cast<const AstComponentType&>(node);
83  return name == other.name && typeParams == other.typeParams;
84  }
85 
86 private:
87  /** component name */
88  std::string name;
89 
90  /** component type parameters */
91  std::vector<AstTypeIdentifier> typeParams;
92 };

159  void print(std::ostream& os) const override {
160  os << ".comp " << *componentType << " ";
161  if (!baseComponents.empty()) {

163  }
164  os << "{\n";
165  if (!components.empty()) {
166  os << join(components, "\n", print_deref<std::unique_ptr<AstComponent>>()) << "\n";
167  }
168  if (!instantiations.empty()) {
169  os << join(instantiations, "\n", print_deref<std::unique_ptr<AstComponentInit>>()) << "\n";
170  }
171  if (!types.empty()) {
172  os << join(types, "\n", print_deref<std::unique_ptr<AstType>>()) << "\n";
173  }
174  if (!relations.empty()) {
175  os << join(relations, "\n", print_deref<std::unique_ptr<AstRelation>>()) << "\n";
176  }
177  if (!overrideRules.empty()) {
178  os << ".override ";
179  os << join(overrideRules, ",") << "\n";
180  }
181  if (!clauses.empty()) {
182  os << join(clauses, "\n\n", print_deref<std::unique_ptr<AstClause>>()) << "\n";
183  }
184  if (!loads.empty()) {
185  os << join(loads, "\n\n", print_deref<std::unique_ptr<AstLoad>>()) << "\n";
186  }
187  if (!printSizes.empty()) {
188  os << join(printSizes, "\n\n", print_deref<std::unique_ptr<AstPrintSize>>()) << "\n";
189  }
190  if (!stores.empty()) {
191  os << join(stores, "\n\n", print_deref<std::unique_ptr<AstStore>>()) << "\n";
192  }

194  os << "}\n";
195  }
