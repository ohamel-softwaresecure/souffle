37 /** Types of relation qualifiers defined as bits in a word */
38 
39 /* relation is read from csv file */
40 #define INPUT_RELATION (0x1)
41 
42 /* relation is written to csv file */

49 #define OVERRIDABLE_RELATION (0x8)

71 /*!
72  * @class Relation
73  * @brief Intermediate representation of a datalog relation

75  * A relation has a name, types of its arguments, qualifier type,

83  void print(std::ostream& os) const override {
84  os << ".decl " << this->getName() << "(";
85  if (!attributes.empty()) {
86  os << attributes[0]->getAttributeName() << ":" << attributes[0]->getTypeName();

198  /** Add a clause to the relation */
199  void addClause(std::unique_ptr<AstClause> clause) {
200  assert(clause != nullptr && "Undefined clause");
201  assert(clause->getHead() != nullptr && "Undefined head of the clause");
202  assert(clause->getHead()->getName() == name &&
203  "Name of the atom in the head of the clause and the relation do not match");
204  clauses.push_back(std::move(clause));
205  }
206 
207  /** Removes the given clause from this relation */
208  bool removeClause(const AstClause* clause) {
209  for (auto it = clauses.begin(); it != clauses.end(); ++it) {
210  if (**it == *clause) {
211  clauses.erase(it);
212  return true;
213  }
214  }
215  return false;
216  }

223  AstRelation* clone() const override {
224  auto res = new AstRelation();
225  res->name = name;
226  res->setSrcLoc(getSrcLoc());
227  for (const auto& cur : attributes) {
228  res->attributes.emplace_back(cur->clone());
229  }
230  for (const auto& cur : clauses) {
231  res->clauses.emplace_back(cur->clone());
232  }
233  res->qualifier = qualifier;
234  return res;
235  }

238  for (auto& cur : attributes) {

242  cur = map(std::move(cur));
243  }
244  }

246  std::vector<const AstNode*> getChildNodes() const override {
247  std::vector<const AstNode*> res;
248  for (const auto& cur : attributes) {
249  res.push_back(cur.get());
250  }
251  for (const auto& cur : clauses) {
252  res.push_back(cur.get());
253  }
254  return res;
255  }

257 protected:
258  /** Implements the node comparison for this node type */
259  bool equal(const AstNode& node) const override {
260  assert(nullptr != dynamic_cast<const AstRelation*>(&node));
261  const auto& other = static_cast<const AstRelation&>(node);
262  return name == other.name && equal_targets(attributes, other.attributes) &&
263  equal_targets(clauses, other.clauses);
264  }

281  /** Datastructure to use for this relation */
