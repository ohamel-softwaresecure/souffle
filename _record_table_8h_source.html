31 /**
32  * A bidirectional mapping between tuples and reference indices.
33  */
34 class RecordMap {
35  /** The arity of the stored tuples */
36  const size_t arity;
37 
38  /** The mapping from tuples to references/indices */
39  std::map<std::vector<RamDomain>, RamDomain> recordToIndex;
40 
41  /** The mapping from indices to tuples */
42  std::vector<std::vector<RamDomain>> indexToRecord;
43 
44 public:
45  explicit RecordMap(size_t arity) : arity(arity), indexToRecord(1) {} // note: index 0 element left free
46 
47  /**
48  * Pack the given vector -- create a new reference in necessary.
49  */
50  RamDomain pack(const std::vector<RamDomain>& vector) {
51  RamDomain index;
52 #pragma omp critical(record_pack)
53  {
54  auto pos = recordToIndex.find(vector);
55  if (pos != recordToIndex.end()) {
56  index = pos->second;
57  } else {
58 #pragma omp critical(record_unpack)
59  {
60  indexToRecord.push_back(vector);
61  index = indexToRecord.size() - 1;
62  recordToIndex[vector] = index;
63 
64  // assert that new index is smaller than the range
65  assert(index != std::numeric_limits<RamDomain>::max());
66  }
67  }
68  }
69  return index;
70  }
71 
72  /**
73  * Packs the given tuple -- and may create a new reference if necessary.
74  */
75  RamDomain pack(const RamDomain* tuple) {
76  std::vector<RamDomain> tmp(arity);
77  for (size_t i = 0; i < arity; i++) {
78  tmp[i] = tuple[i];
79  }
80 
81  return pack(tmp);
82  }
83 
84  /**
85  * Obtains a pointer to the tuple addressed by the given index.
86  */
87  RamDomain* unpack(RamDomain index) {
88  RamDomain* res;
89 
90 #pragma omp critical(record_unpack)
91  res = &(indexToRecord[index][0]);
92 
93  return res;
94  }
95 
96  const RamDomain* unpack(RamDomain index) const {
97  const RamDomain* res;
98 
99 #pragma omp critical(record_unpack)
100  res = &(indexToRecord[index][0]);
101 
102  return res;
103  }
104 };

158  ram::Tuple<RamDomain, Arity> tuple;
