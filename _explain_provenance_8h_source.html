128 /** utility function to split a string */
129 inline std::vector<std::string> split(const std::string& s, char delim, int times = -1) {
130  std::vector<std::string> v;
131  std::stringstream ss(s);
132  std::string item;
133 
134  while ((times > 0 || times <= -1) && std::getline(ss, item, delim)) {
135  v.push_back(item);
136  times--;
137  }
138 
139  if (ss.peek() != EOF) {
140  std::string remainder;
141  std::getline(ss, remainder);
142  v.push_back(remainder);
143  }
144 
145  return v;
146 }
