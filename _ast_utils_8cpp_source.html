49 std::set<const AstRelation*> getBodyRelations(const AstClause* clause, const AstProgram* program) {
50  std::set<const AstRelation*> bodyRelations;
51  for (const auto& lit : clause->getBodyLiterals()) {
52  visitDepthFirst(
53  *lit, [&](const AstAtom& atom) { bodyRelations.insert(getAtomRelation(&atom, program)); });
54  }
55  for (const auto& arg : clause->getHead()->getArguments()) {
56  visitDepthFirst(
57  *arg, [&](const AstAtom& atom) { bodyRelations.insert(getAtomRelation(&atom, program)); });
58  }
59  return bodyRelations;

62 size_t getClauseNum(const AstProgram* program, const AstClause* clause) {
63  // TODO (azreika): This number might change between the provenance transformer and the AST->RAM
64  // translation. Might need a better way to assign IDs to clauses... (see PR #1288).
65  const AstRelation* rel = program->getRelation(clause->getHead()->getName());
66  assert(rel != nullptr && "clause relation does not exist");
67 
68  size_t clauseNum = 1;
69  for (const auto* cur : rel->getClauses()) {
70  bool isFact = cur->getBodyLiterals().empty();
71  if (cur == clause) {
72  return isFact ? 0 : clauseNum;
73  }
74 
75  if (!isFact) {
76  clauseNum++;
77  }
78  }
79  assert(false && "clause does not exist");
80 }

125 bool isFact(const AstClause& clause) {
126  // there must be a head
127  if (clause.getHead() == nullptr) {
128  return false;
129  }
130  // there must not be any body clauses
131  if (!clause.getBodyLiterals().empty()) {
132  return false;
133  }
134 
135  // and there are no aggregates
136  bool hasAggregates = false;
137  visitDepthFirst(*clause.getHead(), [&](const AstAggregator& cur) { hasAggregates = true; });
138  return !hasAggregates;
139 }

155 AstClause* reorderAtoms(const AstClause* clause, const std::vector<unsigned int>& newOrder) {
156  // Find all atom positions
157  std::vector<unsigned int> atomPositions;
158  std::vector<AstLiteral*> bodyLiterals = clause->getBodyLiterals();
159  for (unsigned int i = 0; i < bodyLiterals.size(); i++) {
160  if (dynamic_cast<AstAtom*>(bodyLiterals[i]) != nullptr) {
161  atomPositions.push_back(i);
162  }
163  }
164 
165  // Validate given order
166  assert(newOrder.size() == atomPositions.size());
167  std::vector<unsigned int> nopOrder;
168  for (unsigned int i = 0; i < atomPositions.size(); i++) {
169  nopOrder.push_back(i);
170  }
171  assert(std::is_permutation(nopOrder.begin(), nopOrder.end(), newOrder.begin()));
172 
173  // Create a new clause with the given atom order, leaving the rest unchanged
174  AstClause* newClause = cloneHead(clause);
175  unsigned int currentAtom = 0;
176  for (unsigned int currentLiteral = 0; currentLiteral < bodyLiterals.size(); currentLiteral++) {
177  AstLiteral* literalToAdd = bodyLiterals[currentLiteral];
178  if (dynamic_cast<AstAtom*>(literalToAdd) != nullptr) {
179  // Atoms should be reordered
180  literalToAdd = bodyLiterals[atomPositions[newOrder[currentAtom++]]];
181  }
182  newClause->addToBody(std::unique_ptr<AstLiteral>(literalToAdd->clone()));
183  }
184 
185  return newClause;
186 }
