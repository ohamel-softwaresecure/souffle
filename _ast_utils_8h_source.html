57 /**
58  * Returns literals of a particular type in the body of a clause.
59  *
60  * @param the clause
61  * @return vector of body literals of the specified type
62  */
63 // TODO (azreika): add caching
64 template <typename T, typename C>
65 std::vector<T*> getBodyLiterals(const C& clause) {
66  std::vector<T*> res;
67  for (auto& lit : clause.getBodyLiterals()) {
68  if (T* t = dynamic_cast<T*>(lit)) {
69  res.push_back(t);
70  }
71  }
72  return res;
73 }
