91  struct Location {
92  int identifier{};
93  int element{};
94  std::unique_ptr<RamRelationReference> relation{nullptr};
95 
96  Location() = default;
97 
98  Location(int ident, int elem, std::unique_ptr<RamRelationReference> rel = nullptr)
99  : identifier(ident), element(elem), relation(std::move(rel)) {}
100 
101  Location(const Location& l) : identifier(l.identifier), element(l.element) {
102  if (l.relation != nullptr) {
103  relation = std::unique_ptr<RamRelationReference>(l.relation->clone());
104  }
105  }
106 
107  Location& operator=(Location other) {
108  identifier = other.identifier;
109  element = other.element;
110  relation = std::move(other.relation);
111  return *this;
112  }
113 
114  bool operator==(const Location& loc) const {
115  return identifier == loc.identifier && element == loc.element;
116  }
117 
118  bool operator!=(const Location& loc) const {
119  return !(*this == loc);
120  }
121 
122  bool operator<(const Location& loc) const {
123  return identifier < loc.identifier || (identifier == loc.identifier && element < loc.element);
124  }
125 
126  void print(std::ostream& out) const {
127  out << "(" << identifier << "," << element << ")";
128  }
129 
130  friend std::ostream& operator<<(std::ostream& out, const Location& loc) {
131  loc.print(out);
132  return out;
133  }
134  };

136  /**
137  * A class indexing the location of variables and record
138  * references within a loop nest resulting from the conversion

141  class ValueIndex {
142  /**
143  * The type mapping variables (referenced by their names) to the
144  * locations where they are used.
145  */
146  using variable_reference_map = std::map<std::string, std::set<Location>>;
147 
148  /**
149  * The type mapping record init expressions to their definition points,
150  * hence the point where they get grounded/bound.
151  */
152  using record_definition_map = std::map<const AstRecordInit*, Location>;
153 
154  /**
155  * A map from AstAggregators to storage locations. Note, since in this case
156  * AstAggregators are indexed by their values (not their address) no standard
157  * map can be utilized.
158  */
159  using aggregator_location_map = std::vector<std::pair<const AstAggregator*, Location>>;
160 
161  /** The index of variable accesses */
162  variable_reference_map var_references;
163 
164  /** The index of record definition points */
165  record_definition_map record_definitions;
166 
167  /** The level of a nested ram operation that is handling a given aggregator operation */
168  aggregator_location_map aggregator_locations;
169 
170  public:
171  // -- variables --
172 
173  void addVarReference(const AstVariable& var, const Location& l) {
174  std::set<Location>& locs = var_references[var.getName()];
175  locs.insert(l);
176  }
177 
178  void addVarReference(const AstVariable& var, int ident, int pos,
179  std::unique_ptr<RamRelationReference> rel = nullptr) {
180  addVarReference(var, Location({ident, pos, std::move(rel)}));
181  }
182 
183  bool isDefined(const AstVariable& var) const {
184  return var_references.find(var.getName()) != var_references.end();
185  }
186 
187  const Location& getDefinitionPoint(const AstVariable& var) const {
188  auto pos = var_references.find(var.getName());
189  assert(pos != var_references.end() && "Undefined variable referenced!");
190  return *pos->second.begin();
191  }
192 
193  const variable_reference_map& getVariableReferences() const {
194  return var_references;
195  }
196 
197  // -- records --
198 
199  // - definition -
200 
201  void setRecordDefinition(const AstRecordInit& init, const Location& l) {
202  record_definitions[&init] = l;
203  }
204 
205  void setRecordDefinition(const AstRecordInit& init, int ident, int pos,
206  std::unique_ptr<RamRelationReference> rel = nullptr) {
207  setRecordDefinition(init, Location({ident, pos, std::move(rel)}));
208  }
209 
210  const Location& getDefinitionPoint(const AstRecordInit& init) const {
211  auto pos = record_definitions.find(&init);
212  if (pos != record_definitions.end()) {
213  return pos->second;
214  }
215  assert(false && "Requested location for undefined record!");
216 
217  static Location fail;
218  return fail;
219  }
220 
221  // -- aggregates --
222 
223  void setAggregatorLocation(const AstAggregator& agg, const Location& loc) {
224  aggregator_locations.push_back(std::make_pair(&agg, loc));
225  }
226 
227  const Location& getAggregatorLocation(const AstAggregator& agg) const {
228  // search list
229  for (const auto& cur : aggregator_locations) {
230  if (*cur.first == agg) {
231  return cur.second;
232  }
233  }
234 
235  // fail
236  std::cout << "Lookup of " << &agg << " = " << agg << " failed\n";
237  assert(false && "Requested aggregation operation is not processed!");
238 
239  const static Location fail = Location();
240  return fail;
241  }

243  // -- others --
244 
245  bool isAggregator(const int level) const {
246  // check for aggregator definitions
247  return any_of(aggregator_locations,
248  [&level](const auto& location) { return location.second.identifier == level; });
249  }
250 
251  bool isSomethingDefinedOn(int level) const {
252  // check for variable definitions
253  for (const auto& cur : var_references) {
254  if (cur.second.begin()->identifier == level) {
255  return true;
256  }
257  }
258  // check for record definitions
259  for (const auto& cur : record_definitions) {
260  if (cur.second.identifier == level) {
261  return true;
262  }
263  }
264  // nothing defined on this level
265  return false;
266  }
