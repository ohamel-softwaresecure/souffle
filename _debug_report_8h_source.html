100  void addSection(const std::string& id, std::string title, std::string code) {
101  std::stringstream codeHTML;
102  std::string escapedCode = std::move(code);
103  while (true) {
104  size_t i = escapedCode.find("<");
105  if (i == std::string::npos) {
106  break;
107  }
108  escapedCode.replace(i, 1, "&lt;");
109  }
110  codeHTML << "<pre>" << escapedCode << "</pre>\n";
111  sections.push_back(DebugReportSection(id, std::move(title), {}, codeHTML.str()));
112  }
113 
114  /**
115  * Outputs a complete HTML document to the given stream,
116  * consisting of an index of all of the sections of the report,
117  * followed by the content of each section.
118  */
119  void print(std::ostream& out) const;
