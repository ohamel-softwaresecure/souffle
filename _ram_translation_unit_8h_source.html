49  /** @brief templated method to compute/retrieve an analysis for a translation unit */
50  template <class Analysis>
51  Analysis* getAnalysis() const {
52  std::string name = Analysis::name;
53  auto it = analyses.find(name);
54  if (it == analyses.end()) {
55  // analysis does not exist yet, create instance and run it.
56  auto analysis = std::make_unique<Analysis>(Analysis::name);
57  analysis->run(*this);
58  // Check it hasn't been created by someone else, and insert if not
59  it = analyses.find(name);
60  if (it == analyses.end()) {
61  analyses[name] = std::move(analysis);
62  }
63  }
64  return dynamic_cast<Analysis*>(analyses[name].get());
65  }

67  /** @brief get the set of alive analyses of the translation unit */

71  result.insert(a.second.get());
72  }
73  return result;
74  }
