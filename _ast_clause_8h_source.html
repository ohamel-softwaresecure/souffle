86 
87  /** updates execution order for rule version */
88  void setOrderFor(int version, std::unique_ptr<AstExecutionOrder> plan) {
89  plans[version] = std::move(plan);
90  }
91 
92  /** get orders */
93  std::map<int, const AstExecutionOrder*> getOrders() const {
94  std::map<int, const AstExecutionOrder*> result;
95  for (auto& plan : plans) {
96  result.insert(std::make_pair(plan.first, plan.second.get()));
97  }
98  return result;
99  }
100 
101  AstExecutionPlan* clone() const override {
102  auto res = new AstExecutionPlan();
103  res->setSrcLoc(getSrcLoc());
104  for (auto& plan : plans) {
105  res->setOrderFor(plan.first, std::unique_ptr<AstExecutionOrder>(plan.second->clone()));
106  }
107  return res;
108  }
109 
110  void apply(const AstNodeMapper& map) override {
111  for (auto& plan : plans) {
112  plan.second = map(std::move(plan.second));
113  }
114  }
115 
116  std::vector<const AstNode*> getChildNodes() const override {
117  std::vector<const AstNode*> childNodes;
118  for (auto& plan : plans) {
119  childNodes.push_back(plan.second.get());
120  }
121  return childNodes;
122  }
123 
124 protected:
125  bool equal(const AstNode& node) const override {
126  assert(nullptr != dynamic_cast<const AstExecutionPlan*>(&node));
127  const auto& other = static_cast<const AstExecutionPlan&>(node);
128  return equal_targets(plans, other.plans);
129  }
130 
131 private:

133  std::map<int, std::unique_ptr<AstExecutionOrder>> plans;

136 /**
137  * Intermediate representation of a datalog clause.
138  *
139  * A clause can either be:
140  * - a fact - a clause with no body (e.g., X(a,b))
141  * - a rule - a clause with a head and a body (e.g., Y(a,b) -: X(a,b))
142  *
143  * TODO (azreika): make clause abstract and split into two subclasses: Rule and Fact
144  */
145 class AstClause : public AstNode {
146 public:
147  void print(std::ostream& os) const override {
148  if (head != nullptr) {
149  head->print(os);
150  }
151  if (!bodyLiterals.empty()) {
152  os << " :- \n ";
153  os << join(getBodyLiterals(), ",\n ", print_deref<AstLiteral*>());
154  }
155  os << ".";
156  if (getExecutionPlan() != nullptr) {
157  getExecutionPlan()->print(os);
158  }
159  }
160 
161  /** Add a Literal to the body of the clause */
162  void addToBody(std::unique_ptr<AstLiteral> literal) {
163  bodyLiterals.push_back(std::move(literal));
164  }
165 
166  /** Set the head of clause to @p h */
167  void setHead(std::unique_ptr<AstAtom> h) {
168  assert(!head && "Head is already set");
169  head = std::move(h);
170  }
171 
172  /** Return the atom that represents the head of the clause */
173  AstAtom* getHead() const {
174  return head.get();
175  }
176 
177  /** Obtains a copy of the internally maintained body literals */
178  std::vector<AstLiteral*> getBodyLiterals() const {
179  return toPtrVector(bodyLiterals);
180  }
181 
182  /** Obtains the execution plan associated to this clause or null if there is none */
183  const AstExecutionPlan* getExecutionPlan() const {
184  return plan.get();
185  }
186 
187  /** Updates the execution plan associated to this clause */
188  void setExecutionPlan(std::unique_ptr<AstExecutionPlan> plan) {
189  this->plan = std::move(plan);
190  }
191 
192  /** Resets the execution plan */
193  void clearExecutionPlan() {
194  plan = nullptr;
195  }
196 
197  AstClause* clone() const override {
198  auto res = new AstClause();
199  res->setSrcLoc(getSrcLoc());
200  if (getExecutionPlan() != nullptr) {
201  res->setExecutionPlan(std::unique_ptr<AstExecutionPlan>(plan->clone()));
202  }
203  res->head = (head) ? std::unique_ptr<AstAtom>(head->clone()) : nullptr;
204  for (const auto& lit : bodyLiterals) {
205  res->bodyLiterals.emplace_back(lit->clone());
206  }
207  return res;
208  }
209 
210  void apply(const AstNodeMapper& map) override {
211  head = map(std::move(head));
212  for (auto& lit : bodyLiterals) {
213  lit = map(std::move(lit));
214  }
215  }
216 
217  std::vector<const AstNode*> getChildNodes() const override {
218  std::vector<const AstNode*> res = {head.get()};
219  for (auto& cur : bodyLiterals) {
220  res.push_back(cur.get());
221  }
222  return res;
223  }
224 
225 protected:
226  bool equal(const AstNode& node) const override {
227  assert(nullptr != dynamic_cast<const AstClause*>(&node));
228  const auto& other = static_cast<const AstClause&>(node);
229  return *head == *other.head && equal_targets(bodyLiterals, other.bodyLiterals);
230  }
231 
232  /** The head of the clause */
233  std::unique_ptr<AstAtom> head;
234 
235  /** The literals in the body of this clause */
236  std::vector<std::unique_ptr<AstLiteral>> bodyLiterals;
237 
238  /** The user defined execution plan -- if any */
239  std::unique_ptr<AstExecutionPlan> plan;
240 };
