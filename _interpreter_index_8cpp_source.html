48 bool Order::valid() const {
49  // Check that all indices are in range.
50  for (int i : order) {
51  if (i < 0 || i >= int(order.size())) {
52  return false;
53  }
54  }
55  // Check that there are no duplicates.
56  for (std::size_t i = 0; i < order.size(); i++) {
57  for (std::size_t j = i + 1; j < order.size(); j++) {
58  if (order[i] == order[j]) {
59  return false;
60  }
61  }
62  }
63  return true;
64 }

206  * A generic data structure index adapter handling the boundary
207  * level order conversion as well as iteration through nested
208  * data structures.
209  *
210  * @tparam Structure the structure to be utilized
211  */
212 template <typename Structure>
213 class GenericIndex : public InterpreterIndex {
214 protected:
215  using Entry = typename Structure::element_type;
216  using Hints = typename Structure::operation_hints;
217  static constexpr int Arity = Entry::arity;
218 
219  // the order to be simulated
220  Order order;
221 
222  // the internal data structure
223  Structure data;
224 
225  using iter = typename Structure::iterator;
226 
227  // a source adapter for streaming through data
228  class Source : public Stream::Source {
229  const Order& order;
230 
231  // the begin and end of the stream
232  iter cur;
233  iter end;
234 
235  // an internal buffer for re-ordered elements
236  std::array<Entry, Stream::BUFFER_SIZE> buffer;
237 
238  public:
239  Source(const Order& order, iter begin, iter end)
240  : order(order), cur(std::move(begin)), end(std::move(end)) {}
241 
242  int load(TupleRef* out, int max) override {
243  int c = 0;
244  while (cur != end && c < max) {
245  buffer[c] = order.decode(*cur);
246  out[c] = buffer[c];
247  ++cur;
248  ++c;
249  }
250  return c;
251  }
252 
253  int reload(TupleRef* out, int max) override {
254  int c = 0;
255  max = std::min(max, Stream::BUFFER_SIZE);
256  while (c < max) {
257  out[c] = buffer[c];
258  ++c;
259  }
260  return c;
261  }
262 
263  std::unique_ptr<Stream::Source> clone() override {
264  auto source = std::make_unique<Source>(order, cur, end);
265  source->buffer = this->buffer;
266  return source;
267  }
268  };

270  virtual souffle::range<iter> bounds(const TupleRef& low, const TupleRef& high, Hints& hints) const {
271  Entry a = order.encode(low.asTuple<Arity>());
272  Entry b = order.encode(high.asTuple<Arity>());
273  // Transfer upper_bound to a equivalent lower bound
274  bool fullIndexSearch = true;
275  for (size_t i = Arity; i-- > 0;) {
276  if (a[i] == MIN_RAM_DOMAIN && b[i] == MAX_RAM_DOMAIN) {
277  b[i] = MIN_RAM_DOMAIN;
278  continue;
279  }
280  if (a[i] == b[i]) {
281  b[i] += 1;
282  fullIndexSearch = false;
283  break;
284  }
285  }
286  if (fullIndexSearch) {
287  return {data.begin(), data.end()};
288  }
289  return {data.lower_bound(a, hints), data.lower_bound(b, hints)};
290  }

392 public:
393  /* lexicographical comparison operation on two tuple pointers */
394  struct comparator {
395  const std::vector<int> order;
396 
397  /* constructor to initialize state */
398  comparator(std::vector<int> order) : order(std::move(order)) {}
399 
400  /* comparison function */
401  int operator()(const TupleRef& x, const TupleRef& y) const {
402  for (auto& i : order) {
403  if (x[i] < y[i]) {
404  return -1;
405  }
406  if (x[i] > y[i]) {
407  return 1;
408  }
409  }
410  return 0;
411  }
412 
413  /* less comparison */
414  bool less(const TupleRef& x, const TupleRef& y) const {
415  return operator()(x, y) < 0;
416  }
417 
418  /* equal comparison */
419  bool equal(const TupleRef& x, const TupleRef& y) const {
420  for (auto& i : order) {
421  if (x[i] != y[i]) {
422  return false;
423  }
424  }
425  return true;
426  }
427  };
